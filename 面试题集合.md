## 5. 函数式编程相关
函数也能直接挂载方法
```JS
function sum(...arg) {
  let result = arg.reduce((a, r) => a + r);
  function sum1(...arg1) {
    result += arg1.reduce((a, r) => a + r);
    return sum1;
  }
  sum1.sumOf = () => {
    return result;
  };
  return sum1;
}
console.log(sum(1, 2).sumOf());
console.log(sum(2, 3)(4).sumOf());
console.log(sum(1)(2)(3)(4).sumOf());
```
