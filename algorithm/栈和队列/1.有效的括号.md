## 1.有效的括号
> https://leetcode-cn.com/problems/valid-parentheses/

1. 栈 方案

```js
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
  let Stack = [];
  const map = {
    "(": ")",
    "[": "]",
    "{": "}",
  };
  for (let i = 0; i < s.length; i++) {
    if (["(", "[", "{"].includes(s[i])) {
      Stack.push(s[i]);
    } else if ([")", "]", "}"].includes(s[i])) {
      const bracket = map[Stack.pop(s[i])];
      if (s[i] !== bracket) return false;
    } else {
      return false;
    }
  }
  return Stack.length ? false : true;
};
console.log(isValid("()[]{}"));
console.log(isValid("([)]"));
console.log(isValid("{[]}"));
```
