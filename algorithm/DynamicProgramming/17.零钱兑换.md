/**
 * @param {number[]} coins
 * @param {number} amount
 * @return {number}
 */
var coinChange = function(coins, amount) {
  var dp = new Array(amount + 1).fill(0);
  for (let i = 0; i < dp.length; i++) {
    if (i !== 0) {
      let tempArr = [];
      for (let j = 0; j < coins.length; j++) {
        tempArr.push(i < coins[j] ? Infinity : dp[i - coins[j]]);
      }
      dp[i] = Math.min(...tempArr) + 1;
      // dp[i] =
