## 4.上楼梯问题
```
10楼楼梯
最后一步不是9, 就是8. 10=9+8
```

```Javascript
/**
 * ClimbStairs
 * @param {number} all 多少层楼梯
 */
function ClimbStairs(all) {
  let dp = new Array(all);
  for (let i = 0; i < all; i++) {
    if (i === 0) {
      dp[i] = 1;
    } else if (i === 1) {
      dp[i] = 2;
    } else {
      dp[i] = dp[i - 1] + dp[i - 2];
    }
  }
  return dp;
}
```
