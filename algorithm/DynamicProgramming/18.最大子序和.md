## 18.最大子序和
> https://leetcode-cn.com/problems/maximum-subarray/


```JavaScript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
    let dp = [nums[0]];
    let res = nums[0];
    for (let i = 1; i < nums.length; i++) {
        dp[i] = Math.max(nums[i], nums[i] + dp[i - 1])
        res = Math.max(res, dp[i])
    }
    return res
};
console.log(maxSubArray([-2,1,-3,4,-1,2,1,-5,4]))
```
